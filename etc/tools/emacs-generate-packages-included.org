#+NAME: list-packages
#+begin_src bash :session list-pkgs
lisp_dir="${XDG_CONFIG_HOME:-${HOME}/.config}/emacs/site-lisp"
export lisp_dir

function list-packages {
    escript="${HOME}/.config/emacs/etc/tools/list-use-packages.el"
    local temp="$(mktemp)"
    local elfile="$1"
    shift
    local packages

    if [[ $# -ge 1 ]]; then
	local xtra_pkgs=("$@")
	for pkg in "${xtra_pkgs[@]}"; do
	    echo "$pkg" >>"$temp"
	done
    fi
    
    emacs --script "$escript" "$elfile" >>"$temp"

    mapfile -t packages < <(sort "$temp" | uniq)

    printf ';;; Packages included:\n'
    printf ';; '

    for i in "${!packages[@]}"; do
	if ((i > 0)); then
	    printf ', '
	fi
	printf '%s' "${packages[i]}"
    done
    
    printf '\n'
}
#+end_src
